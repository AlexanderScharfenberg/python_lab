327  }
328  if (t1 == int(Blade::TYPE::VALUE)) {
329  result.m_Data.emplace_back(typename MVEC::Blade::TYPE(int(e_lhs.type)), e_lhs.factor * e_rhs.factor);
330  continue;
331  }
332  vt value;
333  if (t0 == t1 && t0 >= int(Blade::TYPE::E1) && t0 <= int(Blade::TYPE::E3))
334  value = 1.0;
335  else if (t0 == int(Blade::TYPE::E0) && t1 == int(MultiVector<_VType>::Blade::TYPE::EINF))
336  value = -1.0;

338  value = -1.0;

341

343  }

346  // we can use
347  // a . (b ^ c) = (a . b) * c - (a . c) * b
348  // (b ^ c) . a = - a . (b ^ c)

351  vt factor;
352  if (e_lhs.outerProduct.empty()) {

354  b = e_rhs;

356  } else {
